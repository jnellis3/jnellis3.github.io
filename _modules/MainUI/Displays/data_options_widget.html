[docs]class Data_Options_Window(qtw.QMdiSubWindow):



    def __init__(self, parent):
        super().__init__()

        self.mw = parent

        self.setWindowTitle("Data Options")

        mainWidget = qtw.QWidget()
        mainLayout = qtw.QVBoxLayout()
        mainWidget.setLayout(mainLayout)

        self.setWidget(mainWidget)

        #Putting in the button and line for file selection
        fileSelectLayout = qtw.QHBoxLayout()
        mainLayout.addLayout(fileSelectLayout)

        self.fileSelectField = qtw.QLineEdit()
        self.fileSelectField.setText(self.mw.dataLog.fileLocation)
        self.fileSelectField.setReadOnly(True)
        fileSelectLayout.addWidget(self.fileSelectField)

        self.fileSelectBtn = qtw.QPushButton("Select File")
        self.fileSelectBtn.clicked.connect(self.selectFile)
        fileSelectLayout.addWidget(self.fileSelectBtn)

        #Datalog options
        optionsWidget = qtw.QGroupBox("Data Log Options")
        optionsLayout = qtw.QFormLayout()
        optionsWidget.setLayout(optionsLayout)
        mainLayout.addWidget(optionsWidget)

        self.appendDateCheck = qtw.QCheckBox()
        self.appendDateCheck.setChecked(self.mw.dataLog.appendDate)
        optionsLayout.addRow("Append Date to File?", self.appendDateCheck)

        self.autoUpdateCheck = qtw.QCheckBox()
        self.autoUpdateCheck.setChecked(self.mw.dataLog.autoUpdate)
        optionsLayout.addRow("Auto Update Filename Daily?", self.autoUpdateCheck)

        #Adding button for applying changes
        self.applyBtn = qtw.QPushButton("Apply Changes")
        self.applyBtn.clicked.connect(self.applyChanges)

        mainLayout.addWidget(self.applyBtn)


[docs]    def selectFile(self):

        fileName, _ = qtw.QFileDialog.getSaveFileName(self, "Select Data Log File", "./Configurations", "JSON Files (*.json)")
        fileName = fileName.rsplit('.',1)[0] + '.txt'
        self.fileSelectField.setText(fileName)
        return




[docs]    def applyChanges(self):

        self.mw.dataLog.appendDate = bool(self.appendDateCheck.isChecked())
        self.mw.dataLog.autoUpdate = bool(self.autoUpdateCheck.isChecked())
        self.mw.dataLog.fileLocation = self.fileSelectField.text()

        g.DisplayMessage("Data Log Options Updated")

        return
