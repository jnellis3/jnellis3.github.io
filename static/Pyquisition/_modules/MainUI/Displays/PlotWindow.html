    """
    This widget is capable of displaying outputs of channels contained in the 
    activeChannels model. Can display multiple channels.

    :param name:        Name of the widget

    :param configDict:  Dictionary containing the configuration of the widget. If present, the restoreState method is called

    :param chList:      List of channels that can be displayed in the widget. Will likely be the actChW

    :ivar myType:       String containing the type of the widget. This is used to identify the widget when it is saved to a file.

    :ivar channels:     List of channels that can be displayed in the widget. Will likely be the actChW

    :ivar colors:       List of colors that can be used to plot the data. The colors will be cycled through as more channels are added.

    :ivar count:        Number of data points that have been plotted. Used to determine when to clear the plot.

    :ivar range:        Number of data points that will be displayed on the plot at any given time.

    :ivar pointer:      Index of the most recent data point that has been plotted.

    :ivar timer:        Timer that calls the process method at a regular interval.
    """
    def __init__(self, name:str, configDict:dict=None, chList=None):
        super().__init__(name, configDict)
        self.myType = "PlotDisplayWidget"
        self.channels:ChannelWindow= chList
        plotWindow = qtw.QWidget()
        self.colors = ['b','r','g','c','m','y', 'w']

        vLayout = qtw.QVBoxLayout(plotWindow)
        self.topOpts = qtw.QHBoxLayout(plotWindow)

        graphWidget = pg.GraphicsLayoutWidget(self)
        self.plot = graphWidget.ci.addPlot(axisItems={'bottom':pg.DateAxisItem()})
        self.plot.addLegend()
        
        self.channelDropDown = qtw.QComboBox()
        self.channelDropDown.addItems(chList.getChannelNames())
        self.topOpts.addWidget(self.channelDropDown)

        self.add_btn = qtw.QPushButton("Add Channel")
        self.add_btn.clicked.connect(lambda: self.addChannel(None))
        self.removeChannel_btn = qtw.QPushButton('Remove Channel')
        self.color_btn = pg.ColorButton()
        self.color_btn.sigColorChanged.connect(self.changeColorDone)
        self.removeChannel_btn.clicked.connect(lambda: self.removeChannel(None))
        

        self.selectoutput_combo = qtw.QComboBox()
        self.selectoutput_combo.setPlaceholderText("Select Channel Output")

        self.topOpts.addWidget(self.selectoutput_combo)
        self.topOpts.addWidget(self.add_btn)
        self.topOpts.addWidget(self.removeChannel_btn)
        self.topOpts.addWidget(self.color_btn)

        vLayout.addLayout(self.topOpts)
        vLayout.addWidget(graphWidget)

        self.count = 0
        self.range = 300
        self.pointer = 0

        self.timer = qtc.QTimer()
        self.timer.timeout.connect(self.process)
        self.timer.start(100)
        self.setWidget(plotWindow)

        self.channels.treeViewChanged.connect(self.updateChannelList)

        self.currentSelection = self.channelDropDown.currentText()
        if configDict is not None: self.restoreState(configDict)

        self.add_btn.setToolTip("Add " + str(self.channelDropDown.currentText()) + " to the plot")
        self.removeChannel_btn.setToolTip("Remove " + str(self.channelDropDown.currentText()) + " from the plot")
        self.channelDropDownChanged()

[docs]    def restoreState(self, state): 
        for c in state["Inputs"]: 
            try:
                self.addChannel(c)
            except Exception as e:
                global_vars.recordError('Error below: ' + c + ' channel was not found in config file, but was listed as an input for the plot widget.')
                global_vars.recordError(e)




[docs]    def removeChannel(self, ch:str = None): 
        channelName = self.channelDropDown.currentText()
        if ch is not None:
            channelName = ch
        for row in range(self.channels.model.rowCount()):
            if channelName == self.channels.model.item(row).name:
                channel = self.channels.model.item(row)
                channel.signals.outputChanged.disconnect(self.newDataReceived)
                for item in self.plot.listDataItems():
                    if item.name() == channelName:
                        self.plot.removeItem(item)
        return

        channel_name = data[0]
        outputs:dict = data[1]

        #self.outputs[channel_name]['times'].append(outputs.pop('time'))

        for output in outputs:
            try: 
                value = float(outputs[output])
                if output in self.outputs[channel_name].keys():
                    self.outputs[channel_name][output].append(value)
                else:
                    self.outputs[channel_name][output] = [value]
            except: pass # only want to collect outputs that are numbers


        if data[0] in self.inputs.keys():
            self.pointer+=1
            self.outputs[data[0]]['values'].append(data[1][self.outputs[data[0]]['var']])
            self.outputs[data[0]]['times'].append(data[1]['time'])
            self.outputs[data[0]]['line'].setData(self.outputs[data[0]]['times'],self.outputs[data[0]]["values"])

        except Exception as e:

[docs]    def saveState(self):
        configDict = {}
        configDict["Name"] = self.name
        configDict["Type"] = self.myType
        configDict["Inputs"] = []
        for chName in self.inputs.keys():
            configDict["Inputs"].append(str(chName))
        return configDict
