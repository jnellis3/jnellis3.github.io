    def __init__(self, config_file=global_vars.blank_config, config_file_location='./.temp/last_session_config.json'):
        
        super().__init__()

        self.bootSound = SoundEffect('/src/utils_lib/sounds/boot.wav')
        self.bootSound.play()
        self.new_subwindow_sound = SoundEffect('utils_lib/sounds/5.wav')
        self.closeSound = SoundEffect('/src/utils_lib/sounds/11.wav')
        self.config_file = config_file
        self.config_file_location = config_file_location
        self.title = 'Blank Config'


        # Initialize the resource manager. If no instruments are found the system uses a simulator.
        # May want to make this behavior more controllable/intelligent in the future.
        self.rm = pyvisa.ResourceManager()
        sim_lib = ""
        if len(self.rm.list_resources()) < 3:
            self.rm.close()
            sim_lib = os.getcwd() + '/src/Instrumentation/Instrument_Simulators.yaml@sim'
            self.rm = pyvisa.ResourceManager(sim_lib)

        self.instManager = InstrumentModel_Manager(self.rm, sim_lib)
        #Genral Parameters
        self.setWindowTitle('PyQuisition'), self.setGeometry(30,30,1280,800)
        
        #Define the central widget
        self.cw = qtw.QMdiArea(self)
        self.cw.setMinimumSize(600,600), self.setCentralWidget(self.cw)
        self.cw.setDocumentMode(True)
        
        #Creating the dock widgets to diplay the models
        self.actChW = ChannelWindow(self, 'ActiveChannelList', resource_manager=self.rm)
        self.actChW.setWindowTitle('Active Channels')
        self.actChW.setFeatures(qtw.QDockWidget.DockWidgetFeature.NoDockWidgetFeatures), self.actChW.setMinimumWidth(5)

        self.deactChW = ChannelWindow(self, 'DeactiveChannelList', resource_manager=self.rm)
        self.deactChW.setWindowTitle('Deactive Channels')
        self.deactChW.setFeatures(qtw.QDockWidget.DockWidgetFeature.NoDockWidgetFeatures), self.deactChW.setMinimumWidth(5)

        #Adding widgets to the main screen
        self.addDockWidget(qtc.Qt.DockWidgetArea.LeftDockWidgetArea,self.actChW)
        self.addDockWidget(qtc.Qt.DockWidgetArea.LeftDockWidgetArea, self.deactChW)
        
        
        self.paused = False
        self.itemsToUnPause = []
        if self.config_file: 
            self.restoreState()
            with open(self.config_file_location, 'w') as f:
                json.dump(self.config_file, f, indent = 4)

        else: self.dataLog = acc.Accumulator('/.temp/tmp.txt',None,self.actChW)

        

        self.setMenuBar(PyQuisitionMenu(self))



[docs]    def check_config_diffs(self):
        initial_config = json.load(open(self.config_file_location,'r'))
        self.saveState(self.config_file_location)
        final_config = json.load(open(self.config_file_location,'r'))

        diffs = compare(initial_config, final_config)
        if diffs != '{}':
            with open(self.config_file_location.rsplit('.',1)[0] + "_journal.txt", 'a') as f:
                f.write(diffs)
                f.write('\n')

[docs]    def restoreState(self):

        config = self.config_file
        self.title = config['Title']
        self.setWindowTitle(self.title)
        #Setting the accumulator
        self.dataLog = acc.Accumulator('test',config['DataLog'],self.actChW)
        
        #Creating the channel lists
        self.actChW.restoreState(config["ActiveChannels"])
        self.deactChW.restoreState(config["DeactiveChannels"])


        #Creating the display widgets
        for displayDict in config["Plots"]:
            try:
                newPlotClass: DisplayWidget = globals()[displayDict['Type']]
                newDisplay = newPlotClass(displayDict["Name"], displayDict, self.actChW)
            except:
                pass
            
            self.cw.addSubWindow(newDisplay)
            newDisplay.show()
        
        self.cw.tileSubWindows()



[docs]    def saveState(self, file_location):
        
        configDict = {}
        configDict["Title"] = self.title
        configDict["ActiveChannels"] = self.actChW.saveState()
        configDict["DeactiveChannels"] = self.deactChW.saveState()
        configDict["Plots"] = []
        for display in self.cw.subWindowList():
            try:
                newDisplay = display.saveState()
                configDict["Plots"].append(newDisplay)
            except Exception as e:
                global_vars.recordError('Error below: ' + str(type(display)) + ' was not able to be saved.')
                global_vars.recordError(e)

        configDict["DataLog"] = self.dataLog.saveState()
        with open(file_location, 'w') as f:
            json.dump(configDict, f, indent=4)
