    """
    =============================================================================================================================
    This is the channel used to control the attocube instrument.

    It uses an internal thread to execute a loop taking values from the instrument at the rate specified in the control widget

    :param freq:        Frequency of the loop in Hz
    :param averageTime: Time to average over in seconds
    :param averageN:    3dB low-pass filter constant
    :param address:     Address of the instrument
    :param myType:      Type of the channel
    
    =============================================================================================================================
    """
    
    def __init__(self, channelName: str, state: dict = None, parent = None):
        self.freq = 250
        self.averageTime = 1
        self.averageN = 0
        self.address = "192.168.1.1"
        self.myType = "AttoCubeChannel"
        self.instrument = Device(self.address)
        connected = False
        while connected == False:
            try:
                self.instrument.connect()
                connected = True
            except:
                getAddressBox = qtw.QInputDialog()
                getAddressBox.setTextValue(self.address)
                getAddressBox.setLabelText("Unable to connect, please enter Attocube IP Address, type (d) to delete channel")
                getAddressBox.exec()
                self.address = getAddressBox.textValue()
                self.instrument = Device(self.address)
                if self.address == 'd':
                    connected = True
                
        self.children = []
        super().__init__("Attocube", state)
        self.addInput('Instrument', self.instrument)
        self.paused = True
        self.threadPool = qtc.QThreadPool().globalInstance()
        
        self.rawSum = 0
        self.numPoints = 0
        self.setBackground(qtg.QColor(255,0,0))
        self.ax1 = []
        self.ax2 = []

[docs]    def loop(self, progress_callback):
        
        theTime = time.time()
        self.timeToAverage = time.time() + float(self.averageTime)
        while self.paused == False:
            displacements = self.instrument.displacement.getAbsolutePositions()
            self.ax1.append(displacements[1])
            self.ax2.append(displacements[2])
            if theTime > self.timeToAverage:
                ax1 = np.array(self.ax1)
                ax2 = np.array(self.ax2)

                total = ax1+ax2
                self.outputs = {'value': np.mean(total), 'axis1': np.mean(ax1), 'axis2':np.mean(ax2),
                                'axis1STD': np.std(ax1), 'axis2STD': np.std(ax2), 'totalSTD': np.std(total),
                                'numPoints': len(total), 'unit': 'pM',
                                'Ax1 Quality': self.instrument.displacement.getAxisSignalQuality(0)[1],
                                'Ax2 Quality': self.instrument.displacement.getAxisSignalQuality(1)[1],
                                'time': time.time()}
                
                progress_callback.emit(1)
                self.ax1 = []
                self.ax2 = []
                self.timeToAverage = time.time() + self.averageTime
            theTime = time.time()



[docs]    def threadFinished(self):
        print('THREAD FINISHED!')
        if self.paused == False:
            worker  = global_vars.Worker(self.loop)
            worker.signals.finished.connect(self.threadFinished)
            worker.signals.progress.connect(self.process)
            worker.signals.error.connect(self.thread_error)
            self.threadPool.start(worker)

[docs]    def pause_unpause(self):
        super().pause_unpause()
        if self.paused == False:
            worker = global_vars.Worker(self.loop)
            worker.signals.finished.connect(self.threadFinished)
            worker.signals.error.connect(self.thread_error)
            worker.signals.progress.connect(self.process)
            self.threadPool.start(worker)
        else:
            self.threadPool.clear()
            self.numPoints = 0
            self.rawSum = 0

    def __init__(self, channel: AttoCubeChannel, parent: qtw.QMainWindow):
        super().__init__(channel)
        self.mw = parent

        self.frequencyText = qtw.QLineEdit()
        self.averageTime = qtw.QLineEdit()
        self.averageN = qtw.QLineEdit()
        
        self.frequencyText.setText(str(self.channel.freq))
        self.averageTime.setText(str(self.channel.averageTime))
        self.averageN.setText(str(self.channel.averageN))
        
        textEdits = [self.frequencyText, self.averageTime, self.averageN]
        textFields = ['Samples per second', 'Averaging Time (s)', 'Average N']
        for i, item in enumerate(textEdits):
            self.formLayout.addRow(qtw.QLabel(textFields[i]), item)
        
        self.applyBtn = qtw.QPushButton('Apply')
        self.applyBtn.clicked.connect(self.apply)
        self.mainLayout.addWidget(self.applyBtn)
        self.setWidget(self.optionsWidget)
        self.show()
