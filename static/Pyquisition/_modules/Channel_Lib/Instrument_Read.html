    def _start_thread(self):
        """
        If the channel is active and there are no other threads running
        in the channels thread pool, this function will activate.

        Creates a Instrument_Channel_Worker thread
        with the run_procedure function as the target.
        """
        
        if self.paused: return
        if self.threadPool.activeThreadCount() > 0: return

        worker = Instrument_Channel_Worker(self.run_procedure)
        worker.signals.progress.connect(self.process)
        worker.signals.finished.connect(self.procedure_finished)
        worker.signals.error.connect(self.procedure_error)
        
        
        self.threadPool.start(worker)

    def _take_sample(self, commands:dict, num_readings:int, instrument:Instrument_Model, deltaT:float)->dict:
        """
        Will take num_readings using the supplied commands dictionary, will
        return the outputs as a dictionary
        """
        points_taken = 0
        outputs = {}

        for command in commands.keys():
            if command != "param_name":
                outputs[command] = []
        
        outputs["time"] = []

        while points_taken < num_readings:
                
                outputs["time"].append(time.time())

                for command in commands.keys():
    
                    if isinstance(commands[command], property):
    
                        reading = commands[command].fget()
                        
                        outputs[command].append(reading)
    
                points_taken += 1
                time.sleep(deltaT)
        
        for output in outputs.keys():
            try: outputs[output] = np.mean(outputs[output])
            except: 
                try: outputs[output] = outputs[output][-1]
                except Exception as e: 
                    print(e)
                    raise e
        return outputs

            



[docs]    def run_procedure(self, progress_callback = None):
        """
        Used by the thread pool, this function continuously runs until the channel is paused or stopped.
        It will take a sample using the commands dictionary, and then emit the outputs to the main thread.
        """
        while True:

            if self.paused: return

            if "Instrument" not in self.inputs.keys(): return
            else: instrument = self.inputs["Instrument"]

            if self.readCommands is None: return
            else: commands = self.readCommands

            num_points = int(self.points_to_average)
            deltaT = float(self.timerSetting)/1000
            
            outputs = self._take_sample(commands, num_points, instrument, deltaT)
            progress_callback.emit([self.name, outputs])

[docs]    def restoreState(self, state):

        # First restore the generic channel settings
        super().restoreState(state)   

        self.timerSetting = int(state["timerSetting"])
        self.points_to_average = int(state["average_points"])

        # Check if an instrument was connected
        if "Instrument" in state.keys():
            
            # Get the instrument from the instModelManager
            inst_ID = state["Instrument"]["address"]
            self.inputs["Instrument"] = self.myParent.instManager.get_instrument(inst_ID)
            
            #Restore the settings of the instrument if the instManager was able to find it
            if self.inputs["Instrument"]:
                self.inputs["Instrument"].restoreState(state["Instrument"])
            else:
                global_vars.DisplayMessage("Error restoring: " + self.name + "\n" 
                                           + "Could not find instrument with address: " + inst_ID 
                                           + "\n" + "Please reconnect the instrment manually")
            
            self.setting_name = state["setting_name"]
            if isinstance(self.inputs["Instrument"], Instrument_Model):
                self.inst_settings = self.inputs["Instrument"]._settingDicts[self.setting_name]
                self.readCommands = self.inst_settings["value"]

        return

        

[docs]    def saveState(self):

        # First collect the generic channel settings
        config = super().saveState()

        config["timerSetting"] = self.timerSetting
        config["average_points"] = self.points_to_average
        try: 
            config["setting_name"] = self.setting_name
            #self.inst_settings.pop("settings_name")
        except: pass
        # Try restoring the instrument, else skip
        if "Instrument" in self.inputs.keys() and self.inputs["Instrument"] is not None:
            config["Instrument"] = self.inputs["Instrument"].saveState(self.inst_settings)

        return config

        """
        This will set the instrument on the channel, and will update the instrument settings widget
        """
        if "Instrument" not in self.channel.inputs.keys():
            if initializing:
                return
            self.channel.inputs["Instrument"] = ""
        if self.instrField.currentInstrumentSelected() is not self.channel.inputs["Instrument"]:
            self.channel.set_instrument(self.instrField.currentInstrumentSelected())
            try:self.InstrumentSettingsWidget.set_instrument(self.channel.inputs["Instrument"])
            except Exception as e: print(e)

            
            
        
            # Get the settings layouts for the instrument
            self.settings_layout_combo.clear()
            self.settings_layout_combo.addItems(self._get_available_layouts())

        self._resize()

        return
