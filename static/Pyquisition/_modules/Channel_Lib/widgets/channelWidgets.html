    This is a QComboBox that is used to select a channel from the channel window.

    """

    def __init__(self, channelWindow, opts:list = None):
        super().__init__()
        self.channelWindow = channelWindow
        self.channelWindow.treeViewChanged.connect(self.updateList)
        self.updateList()
